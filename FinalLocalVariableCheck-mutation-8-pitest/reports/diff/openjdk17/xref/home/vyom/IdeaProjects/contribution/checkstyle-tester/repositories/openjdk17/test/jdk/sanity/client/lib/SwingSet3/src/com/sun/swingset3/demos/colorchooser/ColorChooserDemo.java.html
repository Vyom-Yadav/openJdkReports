24  package com.sun.swingset3.demos.colorchooser;
25  
26  import java.awt.*;
27  import java.awt.event.ActionEvent;
28  import java.awt.event.ActionListener;
29  import javax.swing.*;
30  import javax.swing.border.EmptyBorder;
31  
32  import com.sun.swingset3.demos.JGridPanel;
33  import com.sun.swingset3.demos.ResourceManager;
34  import com.sun.swingset3.DemoProperties;
35  
36  /**
37   * JColorChooserDemo
38   *
39   * @author Jeff Dinkins
40   * @version 1.1 07/16/99
41   */
42  @DemoProperties(value = "JColorChooser Demo", category = "Choosers", description = "Demonstrates JColorChooser, a component which allows the user to pick a color.", sourceFiles = {
43          "com/sun/swingset3/demos/colorchooser/ColorChooserDemo.java",
44          "com/sun/swingset3/demos/colorchooser/BezierAnimationPanel.java", "com/sun/swingset3/demos/JGridPanel.java",
45          "com/sun/swingset3/demos/ResourceManager.java",
46          "com/sun/swingset3/demos/colorchooser/resources/ColorChooserDemo.properties",
47          "com/sun/swingset3/demos/colorchooser/resources/images/ColorChooserDemo.gif" })
48  public class ColorChooserDemo extends JPanel {
49  
50      private static final ResourceManager resourceManager = new ResourceManager(ColorChooserDemo.class);
51  
52      private final BezierAnimationPanel bezAnim = new BezierAnimationPanel();
53  
54      public static final String BACKGROUND = resourceManager.getString("ColorChooserDemo.background");
55      public static final String GRADIENT_1 = resourceManager.getString("ColorChooserDemo.grad_a");
56      public static final String GRADIENT_2 = resourceManager.getString("ColorChooserDemo.grad_b");
57      public static final String PERIMETER = resourceManager.getString("ColorChooserDemo.outer_line");
58      public static final String CHOOSER_TITLE = resourceManager.getString("ColorChooserDemo.chooser_title");
59  
60      private final JButton outerColorButton = new JButton(PERIMETER);
61  
62      private final JButton backgroundColorButton = new JButton(BACKGROUND);
63  
64      private final JButton gradientAButton = new JButton(GRADIENT_1);
65  
66      private final JButton gradientBButton = new JButton(GRADIENT_2);
67  
68      public static final String DEMO_TITLE = ColorChooserDemo.class.getAnnotation(DemoProperties.class).value();
69  
70      /**
71       * main method allows us to run as a standalone demo.
72       */
73      public static void main(String[] args) {
74          JFrame frame = new JFrame(ColorChooserDemo.class.getAnnotation(DemoProperties.class).value());
75  
76          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
77          frame.getContentPane().add(new ColorChooserDemo());
78          frame.setPreferredSize(new Dimension(800, 600));
79          frame.pack();
80          frame.setLocationRelativeTo(null);
81          frame.setVisible(true);
82      }
83  
84      /**
85       * ColorChooserDemo Constructor
86       */
87      public ColorChooserDemo() {
88          setLayout(new BorderLayout());
89  
90          outerColorButton.setIcon(new ColorSwatch(BezierAnimationPanel.BezierColor.OUTER));
91  
92          backgroundColorButton.setIcon(new ColorSwatch(BezierAnimationPanel.BezierColor.BACKGROUND));
93  
94          gradientAButton.setIcon(new ColorSwatch(BezierAnimationPanel.BezierColor.GRADIENT_A));
95  
96          gradientBButton.setIcon(new ColorSwatch(BezierAnimationPanel.BezierColor.GRADIENT_B));
97  
98          ActionListener l = new ActionListener() {
99              public void actionPerformed(ActionEvent e) {
100                 JButton button = (JButton) e.getSource();
101 
102                 final BezierAnimationPanel.BezierColor bezierColor = ((ColorSwatch) button.getIcon()).getBezierColor();
103 
104                 Color current = bezAnim.getBezierColor(bezierColor);
105 
106                 final JColorChooser chooser = new JColorChooser(current != null ? current : Color.WHITE);
107 
108                 ActionListener colorChooserListener = new ActionListener() {
109                     public void actionPerformed(ActionEvent ae) {
110                         bezAnim.setBezierColor(bezierColor, chooser.getColor());
111                     }
112                 };
113 
114                 JDialog dialog = JColorChooser.createDialog(ColorChooserDemo.this, CHOOSER_TITLE, true, chooser,
115                         colorChooserListener, null);
116 
117                 dialog.setVisible(true);
118             }
119         };
120 
121         outerColorButton.addActionListener(l);
122         backgroundColorButton.addActionListener(l);
123         gradientAButton.addActionListener(l);
124         gradientBButton.addActionListener(l);
125 
126         // Add control buttons
127         JPanel buttonPanel = new JPanel(new GridLayout(1, 4, 15, 0));
128 
129         buttonPanel.add(backgroundColorButton);
130         buttonPanel.add(gradientAButton);
131         buttonPanel.add(gradientBButton);
132         buttonPanel.add(outerColorButton);
133 
134         // Add everything to the panel
135         JGridPanel pnContent = new JGridPanel(1, 0, 1);
136 
137         pnContent.cell(buttonPanel, JGridPanel.Layout.CENTER).cell(bezAnim);
138 
139         pnContent.setBorder(new EmptyBorder(10, 0, 0, 0));
140 
141         add(pnContent);
142     }
143 
144     private class ColorSwatch implements Icon {
145         private final BezierAnimationPanel.BezierColor bezierColor;
146 
147         public ColorSwatch(BezierAnimationPanel.BezierColor bezierColor) {
148             this.bezierColor = bezierColor;
149         }
150 
151         public int getIconWidth() {
152             return 11;
153         }
154 
155         public int getIconHeight() {
156             return 11;
157         }
158 
159         public BezierAnimationPanel.BezierColor getBezierColor() {
160             return bezierColor;
161         }
162 
163         public void paintIcon(Component c, Graphics g, int x, int y) {
164             g.setColor(Color.black);
165             g.fillRect(x, y, getIconWidth(), getIconHeight());
166             g.setColor(bezAnim.getBezierColor(bezierColor));
167             g.fillRect(x + 2, y + 2, getIconWidth() - 4, getIconHeight() - 4);
168         }
169     }
