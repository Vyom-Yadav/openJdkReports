58  
59      /**
60       * DTDEvent instances constructed via event reader are missing the notation
61       * and entity declaration information
62       */
63      @Test
64      public void testDTDEvent() {
65          String XML = "<?xml version='1.0' ?>" + "<!DOCTYPE root [\n" + "<!ENTITY intEnt 'internal'>\n" + "<!ENTITY extParsedEnt SYSTEM 'url:dummy'>\n"
66                  + "<!NOTATION notation PUBLIC 'notation-public-id'>\n" + "<!NOTATION notation2 SYSTEM 'url:dummy'>\n"
67                  + "<!ENTITY extUnparsedEnt SYSTEM 'url:dummy2' NDATA notation>\n" + "]>" + "<root />";
68  
69          try {
70              XMLEventReader er = getReader(XML);
71              XMLEvent evt = er.nextEvent(); // StartDocument
72              evt = er.nextEvent(); // DTD
73              if (evt.getEventType() != XMLStreamConstants.DTD) {
74                  Assert.fail("Expected DTD event");
75              }
76              DTD dtd = (DTD) evt;
77              List entities = dtd.getEntities();
78              if (entities == null) {
79                  Assert.fail("No entity found. Expected 3.");
80              } else {
81                  Assert.assertEquals(entities.size(), 3);
82              }
83              // Let's also verify they are all of right type...
84              testListElems(entities, EntityDeclaration.class);
85  
86              List notations = dtd.getNotations();
87              if (notations == null) {
88                  Assert.fail("No notation found. Expected 2.");
89              } else {
90                  Assert.assertEquals(notations.size(), 2);
91              }
92              // Let's also verify they are all of right type...
93              testListElems(notations, NotationDeclaration.class);
94          } catch (Exception e) {
95              Assert.fail(e.getMessage());
96          }
97      }
98  
99      private XMLEventReader getReader(String XML) throws Exception {
100         inputFactory = XMLInputFactory.newInstance();
101 
102         // Check if event reader returns the correct event
103         XMLEventReader er = inputFactory.createXMLEventReader(new StringReader(XML));
104         return er;
105     }
106 
107 
108     private void testListElems(List l, Class expType) {
109         Iterator it = l.iterator();
110         while (it.hasNext()) {
111             Object o = it.next();
112             Assert.assertNotNull(o);
113             Assert.assertTrue(expType.isAssignableFrom(o.getClass()));
114         }
115     }
