24  package transform;
25  
26  import static jaxp.library.JAXPTestUtilities.USER_DIR;
27  import static jaxp.library.JAXPTestUtilities.runWithTmpPermission;
28  
29  import java.io.File;
30  import java.io.StringReader;
31  import java.util.PropertyPermission;
32  
33  import javax.xml.transform.Transformer;
34  import javax.xml.transform.TransformerFactory;
35  import javax.xml.transform.stream.StreamResult;
36  import javax.xml.transform.stream.StreamSource;
37  
38  import org.testng.Assert;
39  import org.testng.annotations.Listeners;
40  import org.testng.annotations.Test;
41  
42  /*
43   * @test
44   * @bug 6216226
45   * @library /javax/xml/jaxp/libs /javax/xml/jaxp/unittest
46   * @run testng/othervm -DrunSecMngr=true -Djava.security.manager=allow transform.Bug6216226Test
47   * @run testng/othervm transform.Bug6216226Test
48   * @summary Test StreamResult(File) is closed after transform().
49   */
50  @Listeners({jaxp.library.FilePolicy.class})
51  public class Bug6216226Test {
52  
53      @Test
54      public final void test() {
55          try {
56              File test = new File(USER_DIR + "bug6216226.txt");
57              TransformerFactory tf = TransformerFactory.newInstance();
58              Transformer xformer = tf.newTransformer();
59              StringReader st = new StringReader("<?xml version=\"1.0\" encoding=\"UTF-8\"?><doc></doc>");
60              StreamSource s = new StreamSource(st);
61              StreamResult r = runWithTmpPermission(() -> new StreamResult(test), new PropertyPermission("user.dir", "read"));
62              xformer.transform(s, r);
63              if (!test.delete()) {
64                  Assert.fail("cannot delete file: " + test.getPath());
65              }
66          } catch (Exception e) {
67              // unexpected failure
68              e.printStackTrace();
69              Assert.fail(e.toString());
70          }
71      }
72  }
