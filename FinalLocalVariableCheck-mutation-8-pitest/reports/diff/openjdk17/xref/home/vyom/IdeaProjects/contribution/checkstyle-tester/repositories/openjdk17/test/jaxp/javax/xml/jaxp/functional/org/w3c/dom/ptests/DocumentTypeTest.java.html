24  package org.w3c.dom.ptests;
25  
26  import static org.testng.Assert.assertEquals;
27  import static org.w3c.dom.ptests.DOMTestUtil.createDOM;
28  
29  import org.testng.Assert;
30  import org.testng.annotations.Listeners;
31  import org.testng.annotations.Test;
32  import org.w3c.dom.DocumentType;
33  import org.w3c.dom.NamedNodeMap;
34  
35  /*
36   * @test
37   * @library /javax/xml/jaxp/libs
38   * @run testng/othervm -DrunSecMngr=true -Djava.security.manager=allow org.w3c.dom.ptests.DocumentTypeTest
39   * @run testng/othervm org.w3c.dom.ptests.DocumentTypeTest
40   * @summary Test DocumentType
41   */
42  @Listeners({jaxp.library.FilePolicy.class})
43  public class DocumentTypeTest {
44  
45      /*
46       * Test testGetEntities method, and verify the entity items.
47       */
48      @Test
49      public void testGetEntities() throws Exception {
50          DocumentType documentType = createDOM("DocumentType01.xml").getDoctype();
51          NamedNodeMap namedNodeMap = documentType.getEntities();
52          // should return both external and internal. Parameter entities are not
53          // contained. Duplicates are discarded.
54          assertEquals(namedNodeMap.getLength(), 3);
55          assertEquals(namedNodeMap.item(0).getNodeName(), "author");
56          assertEquals(namedNodeMap.item(1).getNodeName(), "test");
57          assertEquals(namedNodeMap.item(2).getNodeName(), "writer");
58      }
59  
60      /*
61       * Test getNotations method, and verify the notation items.
62       */
63      @Test
64      public void testGetNotations() throws Exception {
65          DocumentType documentType = createDOM("DocumentType03.xml").getDoctype();
66          NamedNodeMap nm = documentType.getNotations();
67          assertEquals(nm.getLength(), 2); // should return 2 because the notation
68                                           // name is repeated and
69                                           // it considers only the first
70                                           // occurence
71          assertEquals(nm.item(0).getNodeName(), "gs");
72          assertEquals(nm.item(1).getNodeName(), "name");
73      }
74  
75      /*
76       * Test getName method.
77       */
78      @Test
79      public void testGetName() throws Exception {
80          DocumentType documentType = createDOM("DocumentType03.xml").getDoctype();
81          assertEquals(documentType.getName(), "note");
82      }
83  
84      /*
85       * Test getSystemId and getPublicId method.
86       */
87      @Test
88      public void testGetSystemId() throws Exception {
89          DocumentType documentType = createDOM("DocumentType05.xml").getDoctype();
90          assertEquals(documentType.getSystemId(), "DocumentBuilderImpl02.dtd");
91          Assert.assertNull(documentType.getPublicId());
92      }
93  
94  }
