24  package javax.xml.stream.ptests;
25  
26  import static jaxp.library.JAXPTestUtilities.setSystemProperty;
27  import static jaxp.library.JAXPTestUtilities.clearSystemProperty;
28  
29  import static org.testng.Assert.assertNotNull;
30  import static org.testng.Assert.assertNotSame;
31  import static org.testng.Assert.assertSame;
32  import static org.testng.Assert.assertEquals;
33  
34  import javax.xml.stream.XMLEventFactory;
35  
36  import jaxp.library.JAXPDataProvider;
37  
38  import org.testng.annotations.DataProvider;
39  import org.testng.annotations.Listeners;
40  import org.testng.annotations.Test;
41  
42  /*
43   * @test
44   * @bug 8169778
45   * @library /javax/xml/jaxp/libs
46   * @run testng/othervm -DrunSecMngr=true -Djava.security.manager=allow javax.xml.stream.ptests.XMLEventFactoryNewInstanceTest
47   * @run testng/othervm javax.xml.stream.ptests.XMLEventFactoryNewInstanceTest
48   * @summary Tests for XMLEventFactory.newFactory(factoryId , classLoader)
49   */
50  @Listeners({jaxp.library.BasePolicy.class})
51  public class XMLEventFactoryNewInstanceTest {
52  
53      private static final String DEFAULT_IMPL_CLASS =
54          "com.sun.xml.internal.stream.events.XMLEventFactoryImpl";
55      private static final String XMLEVENT_FACTORY_CLASSNAME = DEFAULT_IMPL_CLASS;
56      private static final String XMLEVENT_FACTORY_ID = "javax.xml.stream.XMLEventFactory";
57  
58      @DataProvider(name = "parameters")
59      public Object[][] getValidateParameters() {
60          return new Object[][] {
61              { XMLEVENT_FACTORY_ID, null },
62              { XMLEVENT_FACTORY_ID, this.getClass().getClassLoader() } };
63      }
64  
65      /**
66       * Test if newDefaultFactory() method returns an instance
67       * of the expected factory.
68       * @throws Exception If any errors occur.
69       */
70      @Test
71      public void testDefaultInstance() throws Exception {
72          XMLEventFactory ef1 = XMLEventFactory.newDefaultFactory();
73          XMLEventFactory ef2 = XMLEventFactory.newFactory();
74          assertNotSame(ef1, ef2, "same instance returned:");
75          assertSame(ef1.getClass(), ef2.getClass(),
76                    "unexpected class mismatch for newDefaultFactory():");
77          assertEquals(ef1.getClass().getName(), DEFAULT_IMPL_CLASS);
78      }
79  
80      /*
81       * test for XMLEventFactory.newFactory(java.lang.String factoryClassName,
82       * java.lang.ClassLoader classLoader) factoryClassName points to correct
83       * implementation of javax.xml.stream.XMLEventFactory , should return
84       * newInstance of XMLEventFactory
85       */
86      @Test(dataProvider = "parameters")
87      public void testNewFactory(String factoryId, ClassLoader classLoader) {
88          setSystemProperty(XMLEVENT_FACTORY_ID, XMLEVENT_FACTORY_CLASSNAME);
89          try {
90              XMLEventFactory xef = XMLEventFactory.newFactory(factoryId, classLoader);
91              assertNotNull(xef);
92          } finally {
93              clearSystemProperty(XMLEVENT_FACTORY_ID);
94          }
95      }
96  
97      /*
98       * test for XMLEventFactory.newFactory(java.lang.String factoryClassName,
99       * java.lang.ClassLoader classLoader) factoryClassName is null , should
100      * throw NullPointerException
101      */
102     @Test(expectedExceptions = NullPointerException.class, dataProvider = "new-instance-neg", dataProviderClass = JAXPDataProvider.class)
103     public void testNewFactoryNeg(String factoryId, ClassLoader classLoader) {
104         XMLEventFactory.newFactory(null, null);
105     }
106 
107 }
