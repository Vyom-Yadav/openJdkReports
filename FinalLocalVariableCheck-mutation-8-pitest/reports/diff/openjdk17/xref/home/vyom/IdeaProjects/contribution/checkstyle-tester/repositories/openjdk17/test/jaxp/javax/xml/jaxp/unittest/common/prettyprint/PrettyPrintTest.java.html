255     /*
256      * Bug: 8249867
257      * Verifies the use of the new property "isStandalone" and the
258      * corresponding System property "jdk.xml.isStandalone".
259      *
260      * Bug: 8261209
261      * Verifies that the property takes effect regardless of the settings of
262      * property "pretty-print".
263      */
264     @Test(dataProvider = "setting")
265     public void testIsStandalone_DOMLS(boolean pretty, boolean p, boolean sp,
266             boolean val, String expected)
267             throws Exception {
268         if (sp) {
269             setSystemProperty(SP_JDK_IS_STANDALONE, Boolean.toString(val));
270         }
271         Document document = getDocument();
272         DOMImplementationLS impl = (DOMImplementationLS) document.getImplementation();
273         LSSerializer ser = impl.createLSSerializer();
274         DOMConfiguration config = ser.getDomConfig();
275         if (pretty) {
276             config.setParameter("format-pretty-print", true);
277         }
278         if (p && !sp) {
279             config.setParameter(JDK_IS_STANDALONE, val);
280         }
281         if (sp) {
282             clearSystemProperty(SP_JDK_IS_STANDALONE);
283         }
284         final StringWriter writer = new StringWriter();
285         final LSOutput output = impl.createLSOutput();
286         output.setCharacterStream(writer);
287         ser.write(document, output);
288         String result = writer.toString();
289 
290         Assert.assertEquals(result, expected);
291     }

497     @Test
498     public void testLSSerializerFormatPrettyPrint() {
499 
500         final String XML_DOCUMENT = "<?xml version=\"1.0\" encoding=\"UTF-16\"?>\n"
501                 + "<hello>before child element<child><children/><children/></child>"
502                 + "after child element</hello>";
503         /**
504          * JDK-8035467 no newline in default output
505          */
506         final String XML_DOCUMENT_DEFAULT_PRINT
507                 = "<?xml version=\"1.0\" encoding=\"UTF-16\"?>"
508                 + "<hello>"
509                 + "before child element"
510                 + "<child><children/><children/></child>"
511                 + "after child element</hello>";
512 
513         final String XML_DOCUMENT_PRETTY_PRINT
514                 = "<?xml version=\"1.0\" encoding=\"UTF-16\"?><hello>\n"
515                 + "    before child element\n"
516                 + "    <child>\n"
517                 + "        <children/>\n"
518                 + "        <children/>\n"
519                 + "    </child>\n"
520                 + "    after child element\n"
521                 + "</hello>\n";
522 
523         // it all begins with a Document
524         DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
525         DocumentBuilder documentBuilder = null;
526         try {
527             documentBuilder = documentBuilderFactory.newDocumentBuilder();
528         } catch (ParserConfigurationException parserConfigurationException) {
529             parserConfigurationException.printStackTrace();
530             Assert.fail(parserConfigurationException.toString());
531         }
532         Document document = null;
533 
534         StringReader stringReader = new StringReader(XML_DOCUMENT);
535         InputSource inputSource = new InputSource(stringReader);
536         try {
537             document = documentBuilder.parse(inputSource);
538         } catch (SAXException saxException) {
539             saxException.printStackTrace();
540             Assert.fail(saxException.toString());
541         } catch (IOException ioException) {
542             ioException.printStackTrace();
543             Assert.fail(ioException.toString());
544         }
545 
546         // query DOM Interfaces to get to a LSSerializer
547         DOMImplementation domImplementation = documentBuilder.getDOMImplementation();
548         DOMImplementationLS domImplementationLS = (DOMImplementationLS) domImplementation;
549         LSSerializer lsSerializer = domImplementationLS.createLSSerializer();
550 
551         System.out.println("Serializer is: " + lsSerializer.getClass().getName()
552                 + " " + lsSerializer);
553 
554         // get configuration
555         DOMConfiguration domConfiguration = lsSerializer.getDomConfig();
556 
557         // query current configuration
558         Boolean defaultFormatPrettyPrint
559                 = (Boolean) domConfiguration.getParameter(DOM_FORMAT_PRETTY_PRINT);
560         Boolean canSetFormatPrettyPrintFalse
561                 = (Boolean) domConfiguration.canSetParameter(DOM_FORMAT_PRETTY_PRINT, Boolean.FALSE);
562         Boolean canSetFormatPrettyPrintTrue
563                 = (Boolean) domConfiguration.canSetParameter(DOM_FORMAT_PRETTY_PRINT, Boolean.TRUE);
564 
565         System.out.println(DOM_FORMAT_PRETTY_PRINT + " default/can set false/can set true = "
566                 + defaultFormatPrettyPrint + "/"
567                 + canSetFormatPrettyPrintFalse + "/" + canSetFormatPrettyPrintTrue);
568 
569         // test values
570         assertEquals(defaultFormatPrettyPrint, Boolean.FALSE,
571                 "Default value of " + DOM_FORMAT_PRETTY_PRINT + " should be " + Boolean.FALSE);
572 
573         assertEquals(canSetFormatPrettyPrintFalse, Boolean.TRUE,
574                 "Can set " + DOM_FORMAT_PRETTY_PRINT + " to " + Boolean.FALSE + " should be "
575                 + Boolean.TRUE);
576 
577         assertEquals(canSetFormatPrettyPrintTrue, Boolean.TRUE,
578                 "Can set " + DOM_FORMAT_PRETTY_PRINT + " to " + Boolean.TRUE + " should be "
579                 + Boolean.TRUE);
580 
581         // get default serialization
582         String prettyPrintDefault = lsSerializer.writeToString(document);
583         System.out.println("(default) " + DOM_FORMAT_PRETTY_PRINT + "=="
584                 + (Boolean) domConfiguration.getParameter(DOM_FORMAT_PRETTY_PRINT)
585                 + ": \n\"" + prettyPrintDefault + "\"");
586 
587         assertEquals(prettyPrintDefault, XML_DOCUMENT_DEFAULT_PRINT,
588                 "Invalid serialization with default value, " + DOM_FORMAT_PRETTY_PRINT + "=="
589                 + (Boolean) domConfiguration.getParameter(DOM_FORMAT_PRETTY_PRINT));
590 
591         // configure LSSerializer to not format-pretty-print
592         domConfiguration.setParameter(DOM_FORMAT_PRETTY_PRINT, Boolean.FALSE);
593         String prettyPrintFalse = lsSerializer.writeToString(document);
594         System.out.println("(FALSE) " + DOM_FORMAT_PRETTY_PRINT + "=="
595                 + (Boolean) domConfiguration.getParameter(DOM_FORMAT_PRETTY_PRINT)
596                 + ": \n\"" + prettyPrintFalse + "\"");
597 
598         assertEquals(prettyPrintFalse, XML_DOCUMENT_DEFAULT_PRINT,
599                 "Invalid serialization with FALSE value, " + DOM_FORMAT_PRETTY_PRINT + "=="
600                 + (Boolean) domConfiguration.getParameter(DOM_FORMAT_PRETTY_PRINT));
601 
602         // configure LSSerializer to format-pretty-print
603         domConfiguration.setParameter(DOM_FORMAT_PRETTY_PRINT, Boolean.TRUE);
604         String prettyPrintTrue = lsSerializer.writeToString(document);

715     private Transformer getTransformer(String xsl) throws Exception {
716         TransformerFactory tf = TransformerFactory.newInstance();
717         if (xsl == null) {
718             return tf.newTransformer();
719         }
720 
721         return tf.newTemplates(
722                 new StreamSource(new ByteArrayInputStream(xsl.getBytes())))
723                 .newTransformer();
724     }

726     private Transformer getTransformer(String xsl, boolean html, boolean omit,
727             boolean pretty, boolean p, boolean sp, boolean standalone)
728             throws Exception {
729         if (sp) {
730             setSystemProperty(SP_XSLTC_IS_STANDALONE, standalone ? "yes" : "no");
731         }
732         Transformer transformer = getTransformer(xsl);
733         transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
734         if (omit) {
735             transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
736         }
737         if (html) {
738             transformer.setOutputProperty(OutputKeys.METHOD, "html");
739         }
740         transformer.setOutputProperty(OutputKeys.INDENT, pretty ? "yes" : "no");
741         if (p && !sp) {
742             transformer.setOutputProperty(XSLTC_IS_STANDALONE, standalone ? "yes" : "no");
743         }
744         if (sp) {
745             clearSystemProperty(SP_XSLTC_IS_STANDALONE);
746         }
747         return transformer;
748     }
